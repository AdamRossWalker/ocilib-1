79 inline void Exception::SetWhat(const otext* value) noexcept
80 {
81  if (_what)
82  {
83  delete[] _what;
84  _what = nullptr;
85  }
86 
87  if (!value)
88  {
89  return;
90  }
91 
92 
93 #if defined(OCI_CHARSET_ANSI)
94  const size_t len = strlen(value);
95 #else
96  const size_t len = wcslen(value);
97 #endif
98 
99  _what = new (std::nothrow) otext[len + 1];
100  if (_what)
101  {
102  memcpy(_what, value, (len + 1) * (sizeof(otext)));
103  }
104 }
