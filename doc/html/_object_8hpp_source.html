361 inline ostring Object::ToString() const
362 {
363  if (!IsNull())
364  {
365  unsigned int len = 0;
366 
367  core::Check(OCI_ObjectToText(*this, &len, nullptr));
368 
369  core::ManagedBuffer<otext> buffer(static_cast<size_t>(len + 1));
370 
371  core::Check(OCI_ObjectToText(*this, &len, buffer));
372 
373  return core::MakeString(static_cast<const otext *>(buffer), static_cast<int>(len));
374  }
375 
376  return OCI_STRING_NULL;
377 }
